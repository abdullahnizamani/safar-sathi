from django.db import models
from django.contrib.auth import get_user_model

User = get_user_model()

class Profile(models.Model):
    user = models.OneToOneField(User, on_delete=models.CASCADE, related_name='profile')
    phone_number = models.CharField(max_length=15)
    university = models.CharField(max_length=100, default="FAST National University")
    gender = models.CharField(max_length=10, choices=[('M', 'Male'), ('F', 'Female'), ('O', 'Other')])

class Ride(models.Model):
    STATUS_CHOICES = [('OPEN', 'Open'), ('FULL', 'Full'), ('COMPLETED', 'Completed'), ('CANCELLED', 'Cancelled')]
    
    driver = models.ForeignKey(User, on_delete=models.CASCADE, related_name='drives')
    origin = models.CharField(max_length=255) 
    destination = models.CharField(max_length=255) 
    departure_time = models.DateTimeField()
    available_seats = models.PositiveIntegerField()
    fare = models.PositiveIntegerField() # PKR amount
    transport_type = models.CharField(max_length=50, default="Car") 
    gender_preference = models.CharField(max_length=10, choices=[('ANY', 'Any'), ('MALE', 'Males Only'), ('FEMALE', 'Females Only')], default='ANY')
    status = models.CharField(max_length=15, choices=STATUS_CHOICES, default='OPEN')

class RideRequest(models.Model):
    STATUS_CHOICES = [('PENDING', 'Pending'), ('ACCEPTED', 'Accepted'), ('REJECTED', 'Rejected')]
    
    ride = models.ForeignKey(Ride, on_delete=models.CASCADE, related_name='requests')
    rider = models.ForeignKey(User, on_delete=models.CASCADE, related_name='ride_requests')
    status = models.CharField(max_length=15, choices=STATUS_CHOICES, default='PENDING')
